import java.io.File;
import java.io.FileInputStream;
import java.net.DatagramPacket;
import java.net.DatagramSocket;
import java.net.InetAddress;
import java.net.SocketTimeoutException;
import java.nio.ByteBuffer;
import java.util.Arrays;

public class server {
    private static void sendFile(DatagramSocket server, String filename, InetAddress address, int port) throws Exception {
        server.setSoTimeout(0); // Reset timeout at start
        String path = "C:\\Users\\Admin\\Documents\\GitHub\\UDP_Reliability_Patterns\\" + filename;
        File file = new File(path); // Fixed: Use absolute path
        if (!file.exists() || !file.canRead()) {
            String message = "File not found or not readable!";
            System.out.println("File error: " + filename + " at " + file.getAbsolutePath());
            byte[] errorBytes = message.getBytes();
            sendPacket(server, address, port, -1, errorBytes, true);
            return;
        }

        FileInputStream fis = new FileInputStream(file);
        byte[] fileBuf = new byte[1024];
        int seqNum = 0;

        while (true) {
            int bytesRead = fis.read(fileBuf);
            if (bytesRead == -1) {
                sendPacket(server, address, port, seqNum, new byte[0], true);
                break;
            }

            byte[] data = Arrays.copyOf(fileBuf, bytesRead);
            boolean sent = false;
            int retries = 0;

            while (!sent && retries < 5) {
                sendPacket(server, address, port, seqNum, data, false);
                try {
                    server.setSoTimeout(1000);
                    byte[] ackBuf = new byte[4];
                    DatagramPacket ackPacket = new DatagramPacket(ackBuf, ackBuf.length);
                    server.receive(ackPacket);
                    ByteBuffer bb = ByteBuffer.wrap(ackBuf);
                    int ackSeq = bb.getInt();
                    if (ackSeq == seqNum) {
                        System.out.println("Received ACK for sequence " + seqNum);
                        sent = true;
                        seqNum++;
                    }
                } catch (SocketTimeoutException e) {
                    System.out.println("Timeout for sequence " + seqNum + ", retrying...");
                    retries++;
                }
            }
            if (retries >= 5) {
                System.out.println("Max retries reached for sequence: " + seqNum + ", terminated.");
                break;
            }
        }
        fis.close();
    }

    private static void sendPacket(DatagramSocket server, InetAddress address, int port, int sequenceNumber, byte[] data, boolean isLast) throws Exception {
        ByteBuffer buf = ByteBuffer.allocate(9 + data.length);
        buf.putInt(sequenceNumber);
        buf.putInt(computeChecksum(data));
        buf.put((byte) (isLast ? 1 : 0));
        buf.put(data);
        DatagramPacket packet = new DatagramPacket(buf.array(), buf.array().length, address, port);
        server.send(packet);
        System.out.println("Sent packet with sequence: " + sequenceNumber + (isLast ? " (last)" : ""));
    }

    private static int computeChecksum(byte[] data) {
        int sum = 0;
        for (byte b : data) {
            sum += (b & 0xFF);
        }
        return sum;
    }

    public static void main(String[] args) throws Exception {
        int port = 2207;
        DatagramSocket server = new DatagramSocket(port);
        System.out.println("UDP server is running on port " + port);
        while (true) {
            try {
                byte[] buf = new byte[1024];
                DatagramPacket dp = new DatagramPacket(buf, buf.length);
                server.receive(dp);
                String filename = new String(dp.getData(), 0, dp.getLength()).trim();
                System.out.println("Received request for file: '" + filename + "' from " + dp.getAddress() + ":" + dp.getPort());
                System.out.println("Raw packet data: " + Arrays.toString(Arrays.copyOf(dp.getData(), dp.getLength())));
                if (filename.isEmpty()) {
                    System.out.println("Ignoring empty filename request");
                    continue;
                }
                InetAddress clientAddress = dp.getAddress();
                int clientPort = dp.getPort();
                sendFile(server, filename, clientAddress, clientPort);
            } catch (SocketTimeoutException e) {
                System.out.println("Timeout waiting for request, continuing...");
                continue;
            } catch (Exception e) {
                System.out.println("Error receiving request: " + e.getMessage());
                continue;
            }
        }
    }

import java.net.DatagramSocket;
import java.net.InetAddress;
import java.net.SocketTimeoutException;
import java.nio.ByteBuffer;
import java.util.Arrays;

public class server {
    private static void sendFile(DatagramSocket server, String filename, InetAddress address, int port) throws Exception {
        server.setSoTimeout(0); // Reset timeout at start
        String path = "C:\\Users\\Admin\\Documents\\GitHub\\UDP_Reliability_Patterns\\" + filename;
        File file = new File(path); // Fixed: Use absolute path
        if (!file.exists() || !file.canRead()) {
            String message = "File not found or not readable!";
            System.out.println("File error: " + filename + " at " + file.getAbsolutePath());
            byte[] errorBytes = message.getBytes();
            sendPacket(server, address, port, -1, errorBytes, true);
            return;
        }

        FileInputStream fis = new FileInputStream(file);
        byte[] fileBuf = new byte[1024];
        int seqNum = 0;

        while (true) {
            int bytesRead = fis.read(fileBuf);
            if (bytesRead == -1) {
                sendPacket(server, address, port, seqNum, new byte[0], true);
                break;
            }

            byte[] data = Arrays.copyOf(fileBuf, bytesRead);
            boolean sent = false;
            int retries = 0;

            while (!sent && retries < 5) {
                sendPacket(server, address, port, seqNum, data, false);
                try {
                    server.setSoTimeout(1000);
                    byte[] ackBuf = new byte[4];
                    DatagramPacket ackPacket = new DatagramPacket(ackBuf, ackBuf.length);
                    server.receive(ackPacket);
                    ByteBuffer bb = ByteBuffer.wrap(ackBuf);
                    int ackSeq = bb.getInt();
                    if (ackSeq == seqNum) {
                        System.out.println("Received ACK for sequence " + seqNum);
                        sent = true;
                        seqNum++;
                    }
                } catch (SocketTimeoutException e) {
                    System.out.println("Timeout for sequence " + seqNum + ", retrying...");
                    retries++;
                }
            }
            if (retries >= 5) {
                System.out.println("Max retries reached for sequence: " + seqNum + ", terminated.");
                break;
            }
        }
        fis.close();
    }

    private static void sendPacket(DatagramSocket server, InetAddress address, int port, int sequenceNumber, byte[] data, boolean isLast) throws Exception {
        ByteBuffer buf = ByteBuffer.allocate(9 + data.length);
        buf.putInt(sequenceNumber);
        buf.putInt(computeChecksum(data));
        buf.put((byte) (isLast ? 1 : 0));
        buf.put(data);
        DatagramPacket packet = new DatagramPacket(buf.array(), buf.array().length, address, port);
        server.send(packet);
        System.out.println("Sent packet with sequence: " + sequenceNumber + (isLast ? " (last)" : ""));
    }

    private static int computeChecksum(byte[] data) {
        int sum = 0;
        for (byte b : data) {
            sum += (b & 0xFF);
        }
        return sum;
    }

    public static void main(String[] args) throws Exception {
        int port = 2207;
        DatagramSocket server = new DatagramSocket(port);
        System.out.println("UDP server is running on port " + port);
        while (true) {
            try {
                byte[] buf = new byte[1024];
                DatagramPacket dp = new DatagramPacket(buf, buf.length);
                server.receive(dp);
                String filename = new String(dp.getData(), 0, dp.getLength()).trim();
                System.out.println("Received request for file: '" + filename + "' from " + dp.getAddress() + ":" + dp.getPort());
                System.out.println("Raw packet data: " + Arrays.toString(Arrays.copyOf(dp.getData(), dp.getLength())));
                if (filename.isEmpty()) {
                    System.out.println("Ignoring empty filename request");
                    continue;
                }
                InetAddress clientAddress = dp.getAddress();
                int clientPort = dp.getPort();
                sendFile(server, filename, clientAddress, clientPort);
            } catch (SocketTimeoutException e) {
                System.out.println("Timeout waiting for request, continuing...");
                continue;
            } catch (Exception e) {
                System.out.println("Error receiving request: " + e.getMessage());
                continue;
            }
        }
    }

    private static void sendFile(DatagramSocket server, String filename, InetAddress address, int port) throws Exception {
        server.setSoTimeout(0); // Reset timeout at start
        String path = "C:\\Users\\Admin\\Documents\\GitHub\\UDP_Reliability_Patterns\\" + filename;
        File file = new File(path); // Fixed: Use absolute path
        if (!file.exists() || !file.canRead()) {
            String message = "File not found or not readable!";
            System.out.println("File error: " + filename + " at " + file.getAbsolutePath());
            byte[] errorBytes = message.getBytes();
            sendPacket(server, address, port, -1, errorBytes, true);
            return;
        }

        FileInputStream fis = new FileInputStream(file);
        byte[] fileBuf = new byte[1024];
        int seqNum = 0;

        while (true) {
            int bytesRead = fis.read(fileBuf);
            if (bytesRead == -1) {
                sendPacket(server, address, port, seqNum, new byte[0], true);
                break;
            }

            byte[] data = Arrays.copyOf(fileBuf, bytesRead);
            boolean sent = false;
            int retries = 0;

            while (!sent && retries < 5) {
                sendPacket(server, address, port, seqNum, data, false);
                try {
                    server.setSoTimeout(1000);
                    byte[] ackBuf = new byte[4];
                    DatagramPacket ackPacket = new DatagramPacket(ackBuf, ackBuf.length);
                    server.receive(ackPacket);
                    ByteBuffer bb = ByteBuffer.wrap(ackBuf);
                    int ackSeq = bb.getInt();
                    if (ackSeq == seqNum) {
                        System.out.println("Received ACK for sequence " + seqNum);
                        sent = true;
                        seqNum++;
                    }
                } catch (SocketTimeoutException e) {
                    System.out.println("Timeout for sequence " + seqNum + ", retrying...");
                    retries++;
                }
            }
            if (retries >= 5) {
                System.out.println("Max retries reached for sequence: " + seqNum + ", terminated.");
                break;
            }
        }
        fis.close();
    }

    private static void sendPacket(DatagramSocket server, InetAddress address, int port, int sequenceNumber, byte[] data, boolean isLast) throws Exception {
        ByteBuffer buf = ByteBuffer.allocate(9 + data.length);
        buf.putInt(sequenceNumber);
        buf.putInt(computeChecksum(data));
        buf.put((byte) (isLast ? 1 : 0));
        buf.put(data);
        DatagramPacket packet = new DatagramPacket(buf.array(), buf.array().length, address, port);
        server.send(packet);
        System.out.println("Sent packet with sequence: " + sequenceNumber + (isLast ? " (last)" : ""));
    }

    private static int computeChecksum(byte[] data) {
        int sum = 0;
        for (byte b : data) {
            sum += (b & 0xFF);
        }
        return sum;
    }

    public static void main(String[] args) throws Exception {
        int port = 2207;
        DatagramSocket server = new DatagramSocket(port);
        System.out.println("UDP server is running on port " + port);
        while (true) {
            try {
                byte[] buf = new byte[1024];
                DatagramPacket dp = new DatagramPacket(buf, buf.length);
                server.receive(dp);
                String filename = new String(dp.getData(), 0, dp.getLength()).trim();
                System.out.println("Received request for file: '" + filename + "' from " + dp.getAddress() + ":" + dp.getPort());
                System.out.println("Raw packet data: " + Arrays.toString(Arrays.copyOf(dp.getData(), dp.getLength())));
                if (filename.isEmpty()) {
                    System.out.println("Ignoring empty filename request");
                    continue;
                }
                InetAddress clientAddress = dp.getAddress();
                int clientPort = dp.getPort();
                sendFile(server, filename, clientAddress, clientPort);
            } catch (SocketTimeoutException e) {
                System.out.println("Timeout waiting for request, continuing...");
                continue;
            } catch (Exception e) {
                System.out.println("Error receiving request: " + e.getMessage());
                continue;
            }
        }
    }

        if (!file.exists() || !file.canRead()) {
            String message = "File not found or not readable!";
            System.out.println("File error: " + filename + " at " + file.getAbsolutePath());
            byte[] errorBytes = message.getBytes();
            sendPacket(server, address, port, -1, errorBytes, true);
            return;
        }

        FileInputStream fis = new FileInputStream(file);
        byte[] fileBuf = new byte[1024];
        int seqNum = 0;

        while (true) {
            int bytesRead = fis.read(fileBuf);
            if (bytesRead == -1) {
                sendPacket(server, address, port, seqNum, new byte[0], true);
                break;
            }

            byte[] data = Arrays.copyOf(fileBuf, bytesRead);
            boolean sent = false;
            int retries = 0;

            while (!sent && retries < 5) {
                sendPacket(server, address, port, seqNum, data, false);
                try {
                    server.setSoTimeout(1000);
                    byte[] ackBuf = new byte[4];
                    DatagramPacket ackPacket = new DatagramPacket(ackBuf, ackBuf.length);
                    server.receive(ackPacket);
                    ByteBuffer bb = ByteBuffer.wrap(ackBuf);
                    int ackSeq = bb.getInt();
                    if (ackSeq == seqNum) {
                        System.out.println("Received ACK for sequence " + seqNum);
                        sent = true;
                        seqNum++;
                    }
                } catch (SocketTimeoutException e) {
                    System.out.println("Timeout for sequence " + seqNum + ", retrying...");
                    retries++;
                }
            }
            if (retries >= 5) {
                System.out.println("Max retries reached for sequence: " + seqNum + ", terminated.");
                break;
            }
        }
        fis.close();
    }

    private static void sendPacket(DatagramSocket server, InetAddress address, int port, int sequenceNumber, byte[] data, boolean isLast) throws Exception {
        ByteBuffer buf = ByteBuffer.allocate(9 + data.length);
        buf.putInt(sequenceNumber);
        buf.putInt(computeChecksum(data));
        buf.put((byte) (isLast ? 1 : 0));
        buf.put(data);
        DatagramPacket packet = new DatagramPacket(buf.array(), buf.array().length, address, port);
        server.send(packet);
        System.out.println("Sent packet with sequence: " + sequenceNumber + (isLast ? " (last)" : ""));
    }

    private static int computeChecksum(byte[] data) {
        int sum = 0;
        for (byte b : data) {
            sum += (b & 0xFF);
        }
        return sum;
    }

    public static void main(String[] args) throws Exception {
        int port = 2207;
        DatagramSocket server = new DatagramSocket(port);
        System.out.println("UDP server is running on port " + port);
        while (true) {
            try {
                byte[] buf = new byte[1024];
                DatagramPacket dp = new DatagramPacket(buf, buf.length);
                server.receive(dp);
                String filename = new String(dp.getData(), 0, dp.getLength()).trim();
                System.out.println("Received request for file: '" + filename + "' from " + dp.getAddress() + ":" + dp.getPort());
                System.out.println("Raw packet data: " + Arrays.toString(Arrays.copyOf(dp.getData(), dp.getLength())));
                if (filename.isEmpty()) {
                    System.out.println("Ignoring empty filename request");
                    continue;
                }
                InetAddress clientAddress = dp.getAddress();
                int clientPort = dp.getPort();
                sendFile(server, filename, clientAddress, clientPort);
            } catch (SocketTimeoutException e) {
                System.out.println("Timeout waiting for request, continuing...");
                continue;
            } catch (Exception e) {
                System.out.println("Error receiving request: " + e.getMessage());
                continue;
            }
        }
    }
